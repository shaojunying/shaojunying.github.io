#### 一路快排

```java
public class SingleWayQuickSort {

    void sort(int[] nums) {
        sort(nums, 0, nums.length - 1);
    }
    private void sort(int[] nums, int l, int r) {
        if (l >= r) return;
        int p = partition(nums, l, r);
        sort(nums, l, p - 1);
        sort(nums, p + 1, r);
    }
    private int partition(int[] nums, int l, int r) {
        int pivot = nums[l];
        int i = l;
        // i 指向最后一个小于pivot的元素
        for (int j = l + 1; j <= r; j++) {
            if (nums[j] < pivot){
                swap(nums, ++i, j);
            }
        }
        swap(nums, i, l);
        return i;
    }
    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
```

#### 二路快排

public class TwoWayQuickSort {
    void sort(int[] nums) {
        sort(nums, 0, nums.length - 1);
    }
    private void sort(int[] nums, int l, int r) {
        if (l >= r) return;
        int p = partition(nums, l, r);
        sort(nums, l, p - 1);
        sort(nums, p + 1, r);
    }
    private int partition(int[] nums, int l, int r) {
        int pivot = nums[l];

        int i = l, j = r;
        // [l, i] 小于等于, [j+1,r] 大于
        while (i < j) {
            while (i < j && nums[i + 1] <= pivot) {
                i++;
            }
            while (i < j && nums[j] >= pivot) {
                j--;
            }
            if (i < j) {
                swap(nums, i + 1, j);
                i++;
                j--;
            }
        }
        // i+1 第一个大于 j 第一个小于
        swap(nums, l, j);
        return j;
