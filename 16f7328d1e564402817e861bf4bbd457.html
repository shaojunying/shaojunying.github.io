public class Solution {
    public int reversePairs(int[] nums) {
        // æš´åŠ›è§£æ³• ==> å¯¹äºæ¯ä¸ªå…ƒç´  å‘åéå† è®¡ç®—åŒ…å«è¯¥å…ƒç´ çš„é€†åºå¯¹ä¸ªæ•°
        int res = 0;
        for (int i = 0; i < nums.length; i ++) {
            for (int j = i + 1; j < nums.length; j ++) {
                if (nums[i] > nums[j]) {
                    res ++;
                }
            }
        }
        return res;
    }

public class Solution {
    public int reversePairs(int[] nums) {
        // åˆ†æ²»ç®—æ³•
        // O(NlogN)
        return reversePairs(nums, 0, nums.length - 1);
    }

    private int reversePairs(int[] nums, int l, int r) {
        // [l, r]
        // å…ƒç´ ä¸ªæ•°å°äºç­‰äº1 ä¸ç”¨åšä»»ä½•å¤„ç†
        if (l >= r) {
            return 0;

        // æœ‰å¤šä¸ªå…ƒç´  åˆ†ä¸ºä¸¤éƒ¨åˆ†
        int m = l + (r - l) / 2;
        // [l, m] [m + 1, r]
        int left = reversePairs(nums, l, m);
        int right = reversePairs(nums, m + 1, r);
        int i = l, j = m + 1;
        int mid = 0;
        // å®šä¹‰ä¸€ä¸ªæ–°æ•°ç»„ æ‹·è´åŸæ•°ç»„
        int[] copiedNums = new int[r - l + 1];
        System.arraycopy(nums, l, copiedNums, 0, r - l + 1);
        int k = l;
        while (i <= m && j <= r) {
            if (copiedNums[i - l] > copiedNums[j - l]) {
                mid += m - i + 1;
                nums[k] = copiedNums[j - l];
                j ++;
            }else {
                nums[k] = copiedNums[i - l];
                i ++;
            }
            k ++;
        }
        // å¦‚æœiè¿˜æ²¡èµ°å®Œ ç»§ç»­æ‹·è´
        System.arraycopy(copiedNums, i - l, nums, k, m - i + 1);
        // mid += (r - m) * (m - i + 1);
        return left + mid + right;
    }
    
    public static void main(String[] args) {
        int res = new Solution().reversePairs(new int[]{7,5,6,4});
        System.out.println(res);
    }

}
