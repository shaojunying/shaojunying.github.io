public class SingleWayQuickSort {

    void sort(int[] nums) {
        sort(nums, 0, nums.length - 1);
    }
    private void sort(int[] nums, int l, int r) {
        if (l >= r) return;
        int p = partition(nums, l, r);
        sort(nums, l, p - 1);
        sort(nums, p + 1, r);
    }
    private int partition(int[] nums, int l, int r) {
        int pivot = nums[l];
        int i = l;
        // i æŒ‡å‘æœ€åä¸€ä¸ªå°äºpivotçš„å…ƒç´ 
        for (int j = l + 1; j <= r; j++) {
            if (nums[j] < pivot){
                swap(nums, ++i, j);
            }
        }
        swap(nums, i, l);
        return i;
    }
    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
