前序遍历

栈

class Solution {
    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        // 前序遍历
        Stack<TreeNode> stack = new Stack<>();
        stack.push(root);
        while (!stack.isEmpty()) {
            TreeNode node = stack.pop();
            list.add(node.val);
            if (node.right != null) {
                stack.push(node.right);
            }
            if (node.left != null) {
                stack.push(node.left);
            }
        }
        return list;
    }
}

Morris遍历

class Solution {
    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        // 前序遍历
        TreeNode node = root;
        while (node != null) {
            if (node.left != null) {
                TreeNode rightNode = node.left;
                while (rightNode.right != null && rightNode.right != node) {
                    rightNode = rightNode.right;
                }
                if (rightNode.right == null) {
                    // 左子树还没遍历
                    list.add(node.val);
                    rightNode.right = node;
                    node = node.left;
                }else {
                    // 左子树遍历完毕
                    rightNode.right = null;
                    node = node.right;
                }
            }else {
                // 没有左子树，直接遍历右子树
                list.add(node.val);
                node = node.right;
            }
        }
        return list;
    }
}

中序遍历

栈

class Solution {
    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        // 中序遍历的非递归版本
        Stack<TreeNode> stack = new Stack<>();
        TreeNode node = root;
        while (node != null || !stack.isEmpty()) {
            if (node != null) {
                stack.push(node);
                node = node.left;
            }else {
                // node == null 访问栈顶元素
                node = stack.pop();
                list.add(node.val);
                node = node.right;
            }
        }
        return list;
    }
}

Morris遍历

class Solution {
    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        TreeNode node = root;
        while (node != null) {
            if (node.left == null) {
                // 没有左节点，node就是当前要遍历的节点
                list.add(node.val);
                node = node.right;
            }else {
                // 有左子树，将左子树的最右节点指向当前节点
                TreeNode rightNode = node.left;
                while (rightNode.right != null && rightNode.right != node) {
                    rightNode = rightNode.right;
                }
                if (rightNode.right == node) {
                    // 左子树已经遍历过了
                    list.add(node.val);
                    rightNode.right = null;
                    node = node.right;
                }else { // rightNode.right == null
                    rightNode.right = node;
                    node = node.left;
                }
            }
        }
        return list;
    }
}

后序遍历

栈

class Solution {
    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        TreeNode node = root;
        Stack<TreeNode> stack = new Stack<>();
        TreeNode pre = null;
        while (node != null || !stack.isEmpty()) {
            if (node != null) {
                // 一直遍历左子树
                pre = node;
                stack.push(node);
                node = node.left;
            }else { // node == null
                TreeNode topNode = stack.peek();
                if (topNode.right == pre || topNode.right == null) {
                    // 右子树已遍历完毕，开始遍历当前节点，之后向上返回
                    pre = topNode;
                    stack.pop();
                    list.add(topNode.val);
                }else {
                    // 需要遍历右子树
                    pre = topNode;
                    node = topNode.right;
                }
            }
        }
        return list;
    }
}

Morris遍历

class Solution {
    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        TreeNode node = root;
        while (node != null) {
            if (node.left != null) {
                TreeNode temp = node.left;
                while (temp.right != null && temp.right != node) {
                    temp = temp.right;
                }
                if (temp.right == null) {
                    // 没有遍历过
                    temp.right = node;
                    node = node.left;
                }else {
                    temp.right = null;
                    // 已经遍历完左子树
                    List<Integer> path = getPath(node.left);
                    Collections.reverse(path);
                    list.addAll(path);
                    node = node.right;
                }
            }else {
                // 没有左子树
                node = node.right;
            }
        }
        List<Integer> path = getPath(root);
        Collections.reverse(path);
        list.addAll(path);
        return list;
    }

    private List<Integer> getPath(TreeNode node) {
        List<Integer> list = new ArrayList<>();
        while (node != null) {
            list.add(node.val);
            node = node.right;
        }
        return list;
    }
}
