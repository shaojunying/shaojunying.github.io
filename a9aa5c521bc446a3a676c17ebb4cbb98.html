class Solution {
    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        // å‰åºéå†
        Stack<TreeNode> stack = new Stack<>();
        stack.push(root);
        while (!stack.isEmpty()) {
            TreeNode node = stack.pop();
            list.add(node.val);
            if (node.right != null) {
                stack.push(node.right);
            }
            if (node.left != null) {
                stack.push(node.left);
            }
        }
        return list;
    }

class Solution {
    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        TreeNode node = root;
        while (node != null) {
            if (node.left != null) {
                TreeNode temp = node.left;
                while (temp.right != null && temp.right != node) {
                    temp = temp.right;
                }
                if (temp.right == null) {
                    // æ²¡æœ‰éå†è¿‡
                    temp.right = node;
                    node = node.left;
                }else {
                    temp.right = null;
                    // å·²ç»éå†å®Œå·¦å­æ ‘
                    List<Integer> path = getPath(node.left);
                    Collections.reverse(path);
                    list.addAll(path);
                    node = node.right;
                }
            }else {
                // æ²¡æœ‰å·¦å­æ ‘
                node = node.right;
            }
        }
        List<Integer> path = getPath(root);
        Collections.reverse(path);
        list.addAll(path);
        return list;
    }

    private List<Integer> getPath(TreeNode node) {
        List<Integer> list = new ArrayList<>();
        while (node != null) {
            list.add(node.val);
            node = node.right;
        }
        return list;
    }
}
