import java.util.*;

class Solution {

    /**

    public int fun(int num, int[] candidates) {
        Arrays.sort(candidates);
        int[] nums = toArray(num);
        int i = 0;
        List<Integer> list = new ArrayList<>();
        boolean hasSmall = false;
        for (; i < nums.length; i++) {
            int index = smallOrEqualThan(candidates, nums[i]);
            if (index == -1){
                break;
            }
            if (candidates[index] < nums[i]) {
                hasSmall = true;
            }
            list.add(candidates[index]);
        }
        if (!hasSmall) {
            // [0, i-1]ä¹‹é—´çš„å…ƒç´ éƒ½å¯ä»¥åœ¨candidatesä¸­æ‰¾åˆ°å°äºç­‰äºçš„å…ƒç´ 
            int j = i - 1;
            for (;j >= 0;j --) {
                int index = smallThan(candidates, nums[j]);
                if (index != -1) {
                    list.set(j, candidates[index]);
                    break;
                }else {
                    list.remove(list.size() - 1);
                }
            }
            if (j == -1) {
                Integer[] res = new Integer[nums.length - 1];
                Arrays.fill(res, candidates[candidates.length - 1]);
                return toInt(res);
            }
            // å¯ä»¥å°†æ›¿æ¢ç¬¬jä½
            for (int k = j + 1; k < nums.length; k ++) {
                list.add(candidates[candidates.length - 1]);
            }
        }else {
            for (int j = i; j < nums.length; j++) {
                list.add(candidates[candidates.length - 1]);
            }
        }
        return toInt(list.toArray(new Integer[0]));
    }


    private int toInt(Integer[] nums) {
        int res = 0;
        for (Integer num : nums) {
            res = res * 10 + num;
        }
        return res;
    }

    private int smallThan(int[] nums, int target) {
        for (int i = nums.length - 1; i >= 0; i --) {
            if (nums[i] < target) {
                return i;
            }
        }
        return -1;
    }

    private int smallOrEqualThan(int[] nums, int target) {
        for (int i = nums.length - 1; i >= 0; i --) {
            if (nums[i] <= target) {
                return i;
            }
        }
        return -1;
    }

    private int[] toArray(int num) {
        List<Integer> list = new ArrayList<>();
        while (num != 0) {
            list.add(num % 10);
            num /= 10;
        }
        Collections.reverse(list);
        int[] res = new int[list.size()];
        for (int i = 0; i < list.size(); i++) {
            res[i] = list.get(i);
        }
        return res;
    }


    public static void main(String[] args) {
        System.out.println(new Solution().fun(23121, new int[]{2, 4, 9}));
        System.out.println(new Solution().fun(213121, new int[]{2, 4, 9}));
    }
}
