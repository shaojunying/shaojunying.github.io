class Solution {
    public int[] loudAndRich(int[][] richer, int[] quiet) {

        int n = quiet.length;

        // å…¥é˜Ÿæ‰€æœ‰å…¥åº¦ä¸º0çš„èŠ‚ç‚¹
        Queue<Integer> queue = new ArrayDeque<>();
        for (int i = 0; i < indegrees.length; i ++) {
            if (indegrees[i] == 0) {
                queue.add(i);
            }
        }
        // ä¸ºæ¯ä¸ªèŠ‚ç‚¹èµ‹é»˜è®¤å€¼
        int[] res = new int[n];
        for (int i = 0; i < res.length; i ++) {
            res[i] = i;
        }
        while (!queue.isEmpty()) {
            int element = queue.remove();
            for (int j = 0; j < graph.length; j ++) {
                if (graph[element][j] == 1) {
                    // æœ‰elementæŒ‡å‘jçš„è¾¹

                    // å°è¯•æ›´æ–° é’±æ•°å¤§äºç­‰äºjçš„é’±æ•° ä¸”å®‰é™å€¼æœ€å°çš„å…ƒç´

                    if (quiet[res[element]] < quiet[res[j]]) {
                        res[j] = res[element];
                    }
                    
                    // æ›´æ–°jçš„å…¥åº¦ä¿¡æ¯
                    indegrees[j] --;
                    if (indegrees[j] == 0) {
                        queue.add(j);
                    }
                }
            }
        }
        return res;
    }

    private int[][] convertToGraph(int[][] matrix, int n) {
        int[][] graph = new int[n][n];
        for (int[] pair : matrix) {
            int node1 = pair[0];
            int node2 = pair[1];
            graph[node1][node2] = 1;
        }
        return graph;
    }

    private int[] getIndegree(int[][] graph, int n) {
        int[] indegrees = new int[n];
        for (int i = 0; i < graph.length; i ++) {
            for (int j = 0; j < graph.length; j ++) {
                // æ›´æ–°èŠ‚ç‚¹çš„å…¥åº¦
                if (graph[i][j] == 1) {
                    indegrees[j] ++;
                }
            }
        }
        return indegrees;
    }
